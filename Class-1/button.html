CSS Button Styles & Effects

Basic Buttons

/* Basic button styling */ .btn { display: inline-block; /* Makes the button behave like a block but still remain inline */ padding: 10px 20px; /* Internal spacing (top/bottom, left/right) */ font-size: 16px; cursor: pointer; /* Changes mouse cursor to hand when hovering */ text-align: center; text-decoration: none; /* Removes underline from links */ outline: none; /* Removes outline when clicked/focused */ border: none; background-color: #3498db; color: white; transition: all 0.3s ease; /* Smooth transitions for any property changes */ } /* Change styles on hover */ .btn-basic:hover { background-color: #2980b9; /* Darker blue when hovered */ } /* Rounded corners */ .btn-rounded { border-radius: 25px; /* Creates rounded corners */ } .btn-rounded:hover { transform: scale(1.05); /* Makes button slightly larger on hover */ }

Shadowed Buttons

/* Button with shadow effect */ .btn-shadow { box-shadow: 0 4px 8px rgba(0, 0, 0, 0.2); /* box-shadow: x-offset y-offset blur-radius spread-radius color */ /* rgba allows transparency (0.2 = 20% opacity) */ } .btn-shadow:hover { box-shadow: 0 8px 16px rgba(0, 0, 0, 0.3); /* Bigger, darker shadow on hover */ } /* 3D button effect using shadows and positioning */ .btn-3d { box-shadow: 0 5px 0 #2980b9, 0 5px 10px rgba(0, 0, 0, 0.2); /* Creates two shadows - one solid color for 3D effect, one blurred */ position: relative; top: 0; } .btn-3d:hover { top: 2px; /* Button moves down slightly */ box-shadow: 0 3px 0 #2473a6, 0 3px 6px rgba(0, 0, 0, 0.2); /* Shadow gets smaller to create "pressed" effect */ }

Outlined Buttons

/* Outline button - just border, no background */ .btn-outline { background-color: transparent; /* Transparent background */ color: #3498db; /* Text color same as border */ border: 2px solid #3498db; /* Visible border */ } .btn-outline:hover { background-color: #3498db; /* Fill with color on hover */ color: white; /* Change text color for contrast */ } /* Ghost button - subtle effect with transparent background */ .btn-ghost { background-color: transparent; color: #3498db; border: 2px solid transparent; /* Invisible border initially */ } .btn-ghost:hover { border-color: #3498db; /* Show border on hover */ } /* Using ::after pseudo-element for subtle background effect */ .btn-ghost::after { content: ''; /* Required for pseudo-elements */ position: absolute; top: 0; left: 0; right: 0; bottom: 0; background-color: rgba(52, 152, 219, 0.1); z-index: -1; /* Place behind button content */ transform: scale(0); /* Start at size zero */ transition: transform 0.3s ease; } .btn-ghost:hover::after { transform: scale(1); /* Grow to full size on hover */ }

Gradient Buttons

/* Gradient background from one color to another */ .btn-gradient { background: linear-gradient(to right, #3498db, #9b59b6); /* Creates gradient from left (#3498db) to right (#9b59b6) */ border-radius: 4px; } .btn-gradient:hover { background: linear-gradient(to right, #2980b9, #8e44ad); /* Darker gradient colors on hover */ box-shadow: 0 4px 8px rgba(0, 0, 0, 0.2); } /* Animated gradient that shifts colors continuously */ .btn-gradient-animated { background: linear-gradient( 45deg, #3498db, #9b59b6, #e74c3c, #f1c40f ); background-size: 300% 300%; /* Larger than button for movement */ animation: gradient-shift 5s ease infinite; /* animation: name duration timing-function iteration-count */ } /* Keyframes define the animation steps */ @keyframes gradient-shift { 0% { background-position: 0% 50%; } 50% { background-position: 100% 50%; } 100% { background-position: 0% 50%; } }

Icon Buttons

/* Button with icon using flexbox for alignment */ .btn-icon { display: inline-flex; /* Flex container for alignment */ align-items: center; /* Vertical alignment */ gap: 8px; /* Space between icon and text */ } /* Animated icon button where the icon moves on hover */ .btn-icon-animation .icon { transition: transform 0.3s ease; } .btn-icon-animation:hover .icon { transform: translateX(4px); /* Move icon right on hover */ }

Special Effect Buttons

/* Ripple effect button (requires JavaScript for full effect) */ .btn-ripple { position: relative; overflow: hidden; /* Keeps ripple inside button */ } /* Style for the ripple element created by JavaScript */ .ripple { position: absolute; border-radius: 50%; /* Circle shape */ background-color: rgba(255, 255, 255, 0.6); transform: scale(0); /* Start at size zero */ animation: ripple-effect 0.6s linear; } @keyframes ripple-effect { to { transform: scale(2.5); /* Grow larger */ opacity: 0; /* Fade out */ } } /* Neon glow effect using box-shadow and text-shadow */ .btn-neon { background-color: transparent; color: #3498db; border: 2px solid #3498db; box-shadow: 0 0 5px #3498db, 0 0 5px #3498db inset; /* Two shadows - one outside, one inside (inset) */ text-shadow: 0 0 5px #3498db; /* Text shadow creates glow around text */ } .btn-neon:hover { box-shadow: 0 0 10px #3498db, 0 0 10px #3498db inset; text-shadow: 0 0 10px #3498db; /* Stronger glow effect on hover */ }

Button Colors
